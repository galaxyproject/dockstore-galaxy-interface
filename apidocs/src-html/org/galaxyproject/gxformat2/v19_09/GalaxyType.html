001package org.galaxyproject.gxformat2.v19_09;
002
003import org.galaxyproject.gxformat2.v19_09.utils.ValidationException;
004
005public enum GalaxyType {
006  INTEGER("integer"),
007  TEXT("text"),
008  FILE("File"),
009  DATA("data"),
010  COLLECTION("collection");
011
012  private static String[] symbols = new String[] {"integer", "text", "File", "data", "collection"};
013  private String docVal;
014
015  private GalaxyType(final String docVal) {
016    this.docVal = docVal;
017  }
018
019  public static GalaxyType fromDocumentVal(final String docVal) {
020    for (final GalaxyType val : GalaxyType.values()) {
021      if (val.docVal.equals(docVal)) {
022        return val;
023      }
024    }
025    throw new ValidationException(String.format("Expected one of %s", GalaxyType.symbols, docVal));
026  }
027}
