070    @Override
071    public List<RowData> generateToolsTable(
072        String initialPath,
073        String contents,
074        Map<String, Pair<String, GenericFileType>> indexedFiles) {
075      final Map<String, Object> workflow = loadWorkflow(contents);
076      final Map<String, Object> elements = Cytoscape.getElements(workflow);
077      final List<Map> nodes = (List<Map>) elements.getOrDefault("nodes", Lists.newArrayList());
078      removeStartAndEndNodes(nodes);
079      return nodes.stream()
080          .map(
081              node -> {
082                final RowData rowData = new RowData();
083                final Map<String, Object> nodeData = (Map<String, Object>) node.get("data");
084                rowData.label = (String) nodeData.getOrDefault("label", "");
085                rowData.dockerContainer = (String) nodeData.getOrDefault("docker", "");
086                rowData.filename = (String) nodeData.getOrDefault("run", "");
087                // TODO: get a sane link here when Docker is hooked up
088                try {
089                  rowData.link = new URL((String) node.getOrDefault("repo_link", ""));
090                } catch (MalformedURLException e) {
091                  rowData.link = null;
092                }
093                rowData.rowType = RowType.TOOL;
094                rowData.toolid = (String) nodeData.getOrDefault("id", "");
095                return rowData;
096              })
097          .collect(Collectors.toList());
098    }
099
100    // Remove start and end nodes that were added for DAG
101    private void removeStartAndEndNodes(List<Map> nodes) {
102      nodes.removeIf(
103          node -> {
104            Map data = (Map) node.get("data");
105            String id = (String) data.get("id");
106            return START_ID.equals(id) || END_ID.equals(id);
107          });
108    }

110    @Override
111    public VersionTypeValidation validateWorkflowSet(
112        String initialPath,
113        String contents,
114        Map<String, Pair<String, GenericFileType>> indexedFiles) {
115      final LintContext lintContext = Lint.lint(loadWorkflow(contents));
116      final boolean valid;
117      valid = !lintContext.getFoundErrors();
118      final Map<String, String> messagesAsMap = new HashMap<>();
119      final List<String> validationMessages = lintContext.collectMessages();
120      final StringBuilder builder = new StringBuilder();
121      if (validationMessages.size() == 1) {
122        builder.append(validationMessages.get(0));
123      } else if (validationMessages.size() > 1) {
124        for (final String validationMessage : validationMessages) {
125          builder.append("- ").append(validationMessage).append("\n");
126        }
127      }
128      final String validationMessageMerged = builder.toString();
129      if (validationMessageMerged.length() > 0) {
130        messagesAsMap.put(initialPath, validationMessageMerged);
131      }
132      return new VersionTypeValidation(valid, messagesAsMap);
133    }

152    @Override
153    public Map<String, Pair<String, GenericFileType>> indexWorkflowFiles(
154        final String initialPath, final String contents, final FileReader reader) {
155      Map<String, Pair<String, GenericFileType>> results = new HashMap<>();
156      final Optional<String> testParameterFile = findTestParameterFile(initialPath, reader);
157      testParameterFile.ifPresent(
158          s -> results.put(s, new ImmutablePair<>(s, GenericFileType.TEST_PARAMETER_FILE)));
159      return results;
160    }

162    protected Optional<String> findTestParameterFile(
163        final String initialPath, final FileReader reader) {
164      final int extensionPos = initialPath.lastIndexOf(".");
165      final String base = initialPath.substring(0, extensionPos);
166
167      final Path parent = Paths.get(initialPath).getParent();
168      // listing files is more rate limit friendly (e.g. GitHub counts each 404 "miss" as an API
169      // call,
170      // but listing a directory can be free if previously requested/cached)
171      final Set<String> filenameset =
172          parent == null ? Sets.newHashSet() : Sets.newHashSet(reader.listFiles(parent.toString()));
173
174      for (final String suffix : TEST_SUFFIXES) {
175        for (final String extension : TEST_EXTENSIONS) {
176          final String testFile = base + suffix + extension;
177          if (filenameset.contains(testFile)) {
178            return Optional.of(testFile);
179          }
180        }
181      }
182      return Optional.empty();
183    }

189      } catch (Exception e) {

194    static Map<String, Object> loadWorkflow(final String content) {
195      final Yaml yaml = new Yaml();
196      final Map map = yaml.loadAs(content, Map.class);
197      return (Map<String, Object>) map;
198    }

200    @Override
201    public RecommendedLanguageInterface.WorkflowMetadata parseWorkflowForMetadata(
202        String initialPath,
203        String content,
204        Map<String, Pair<String, GenericFileType>> indexedFiles) {
205      RecommendedLanguageInterface.WorkflowMetadata metadata =
206          new RecommendedLanguageInterface.WorkflowMetadata();
207      if (content != null && !content.isEmpty()) {
208        try {
209          final Map<String, Object> map = loadWorkflow(content);
210          String name = null;
211          try {
212            name = (String) map.get("name");
213          } catch (ClassCastException e) {
214            LOG.debug("\"name:\" is malformed");
215          }
216
217          // FOLLOWING CODE PULLED FROM cwl handler...
218          String label = null;
219          try {
220            label = (String) map.get("label");
221          } catch (ClassCastException e) {
222            LOG.debug("\"label:\" is malformed");
223          }
224
225          String annotation = null;
226          try {
227            annotation = (String) map.get("annotation");
228          } catch (ClassCastException e) {
229            LOG.debug("\"annotation:\" is malformed");
230          }
231
232          // "doc:" added for CWL 1.0
233          String doc = null;
234          if (map.containsKey("doc")) {
235            Object objectDoc = map.get("doc");
236            if (objectDoc instanceof String) {
237              doc = (String) objectDoc;
238            } else if (objectDoc instanceof Map) {
239              Map docMap = (Map) objectDoc;
240              if (docMap.containsKey("$include")) {
241                String enclosingFile = (String) docMap.get("$include");
242                Optional<Pair<String, GenericFileType>> first =
243                    indexedFiles.values().stream()
244                        .filter(pair -> pair.getLeft().equals(enclosingFile))
245                        .findFirst();
246                if (first.isPresent()) {
247                  // No way to fetch this here...
248                  LOG.info(
249                      "$include would have this but reader not passed through, not implemented");
250                  doc = null;
251                }
252              }
253            } else if (objectDoc instanceof List) {
254              // arrays for "doc:" added in CWL 1.1
255              List docList = (List) objectDoc;
256              doc = String.join(System.getProperty("line.separator"), docList);
257            }
258          }
259
260          final String finalChoiceForDescription =
261              ObjectUtils.firstNonNull(doc, annotation, name, label);
262          if (finalChoiceForDescription != null) {
263            metadata.setDescription(finalChoiceForDescription);
264          } else {
265            LOG.info("Description not found!");
266          }
267
268        } catch (YAMLException | NullPointerException | ClassCastException ex) {
269          String message;
270          if (ex.getCause() != null) {
271            // seems to be possible to get underlying cause in some cases
272            message = ex.getCause().toString();
273          } else {
274            // in other cases, the above will NullPointer
275            message = ex.toString();
276          }
277          LOG.info("Galaxy Workflow file is malformed " + message);
278          // CWL parser gets to put validation information in here,
279          // plugin interface doesn't consume the right information though.
280          // https://github.com/dockstore/dockstore/blob/develop/dockstore-webservice/src/main/java/io/dockstore/webservice/languages/CWLHandler.java#L139
281        }
282      }
283      return metadata;
284    }
